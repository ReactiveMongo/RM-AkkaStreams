19 private[akkastream] class DocumentStage[T](
20   cursor: AkkaStreamCursorImpl[T],
21   maxDocs: Int,
22   err: ErrorHandler[Option[T]]
23 )(implicit ec: ExecutionContext)
24     extends GraphStage[SourceShape[T]] {
25 
26   override val toString = "ReactiveMongoDocument"
27   val out: Outlet[T] = Outlet(s"${toString}.out")
28   val shape: SourceShape[T] = SourceShape(out)
29 
30   private val nextResponse = cursor.nextResponse(maxDocs)
31   private val logger = reactivemongo.util.LazyLogger(
32     "reactivemongo.akkastream.DocumentStage"
33   )
34 
35   @inline
36   private def nextR(r: Response): Future[Option[Response]] =
37     nextResponse(ec, r)
38 
39   def createLogic(inheritedAttributes: Attributes): GraphStageLogic =
40     new GraphStageLogic(shape) with OutHandler {
41       private var last = Option.empty[(Response, Iterator[T], Option[T])]
42 
43       @inline private def tailable = cursor.wrappee.tailable
44 
45       private var request: () => Future[Option[Response]] = { () =>
46         cursor.makeRequest(maxDocs).andThen {
47           case Success(r) if (
48             tailable && r.reply.numberReturned > 0
49           ) => onFirst()
50 
51           case Success(_) if (!tailable) => onFirst()
52         }.map(Some(_))
53       }
54 
55       private def kill(): Unit = last.foreach {
56         case (r, _, _) =>
57           try {
58             cursor.wrappee kill r.reply.cursorID
59           } catch {
60             case reason: Throwable =>
61               logger.warn("fails to kill the cursor", reason)
62           }
63 
64           last = None
65       }
66 
67       private def onFirst(): Unit = {
68         request = { () =>
69           last.fold(Future.successful(Option.empty[Response])) {
70             case (lastResponse, _, _) => nextR(lastResponse).andThen {
71               case Success(Some(response)) => last.foreach {
72                 case (lr, _, _) =>
73                   if (lr.reply.cursorID != response.reply.cursorID) try {
74                     cursor.wrappee kill lr.reply.cursorID
75                   } catch {
76                     case reason: Throwable =>
77                       logger.warn("fails to kill the cursor", reason)
78                   }
79               }
80             }
81           }
82         }
83       }
84 
85       private def onFailure(reason: Throwable): Unit = {
86         val previous = last.flatMap(_._3)
87 
88         kill()
89 
90         err(previous, reason) match {
91           case Cursor.Cont(_)     => ()
92           case Cursor.Fail(error) => fail(out, error)
93           case Cursor.Done(_)     => completeStage()
94         }
95       }
96 
97       private def nextD(r: Response, bulk: Iterator[T]): Unit =
98         Try(bulk.next) match {
99           case Failure(reason @ ReplyDocumentIteratorExhaustedException(_)) =>
100             fail(out, reason)
101 
102           case Failure(reason @ CursorOps.Unrecoverable(_)) =>
103             fail(out, reason)
104 
105           case Failure(reason) => err(last.flatMap(_._3), reason) match {
106             case Cont(current @ Some(v)) => {
107               last = Some((r, bulk, current))
108               push(out, v)
109             }
110 
111             case Cont(_) => {}
112 
113             case Done(current @ Some(v)) => {
114               push(out, v)
115               completeStage()
116             }
117 
118             case Done(_)     => completeStage()
119             case Fail(cause) => fail(out, cause)
120           }
121 
122           case Success(v) => {
123             last = Some((r, bulk, Some(v)))
124             push(out, v)
125           }
126         }
127 
128       private val futureCB =
129         getAsyncCallback((response: Try[Option[Response]]) => {
130           response match {
131             case Failure(reason) => onFailure(reason)
132 
133             case Success(resp @ Some(r)) => {
134               if (r.reply.numberReturned == 0) {
135                 if (tailable) onPull()
136                 else {
137                   last = None
138 
139                   completeStage()
140                 }
141               } else {
142                 last = None
143 
144                 val bulkIter = cursor.documentIterator(r).
145                   take(maxDocs - r.reply.startingFrom)
146 
147                 nextD(r, bulkIter)
148               }
149             }
150 
151             case Success(_) if (!tailable) => {
152               kill()
153 
154               last = None
155               completeStage()
156             }
157 
158             case Success(_) => {
159               last = None
160               Thread.sleep(1000) // TODO
161               onPull()
162             }
163           }
164         }).invoke _
165 
166       def onPull(): Unit = last match {
167         case Some((r, bulk, _)) if (bulk.hasNext) =>
168           nextD(r, bulk)
169 
170         case _ if (tailable && !cursor.wrappee.connection.active) => {
171           // Complete tailable source if the connection is no longer active
172           completeStage()
173         }
174 
175         case _ =>
176           request().onComplete(futureCB)
177       }
178 
179       setHandler(out, this)
180     }
