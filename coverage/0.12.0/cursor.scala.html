47 class PlayIterateesCursorImpl[T](val wrappee: Cursor[T])
48     extends PlayIterateesCursor[T] with WrappedCursor[T] {
49   import Cursor.{ Cont, Fail, State }
50 
51   private def errorHandler[T](chan: Concurrent.Channel[T], err: ErrorHandler[Unit]): ErrorHandler[Unit] = {
52     val after: State[Unit] => State[Unit] = {
53       case f @ Fail(e) => {
54         chan.end(e)
55         f
56       }
57 
58       case st => st
59     }
60 
61     (v, e) => after(err(v, e))
62   }
63 
64   override def enumerator(maxDocs: Int = Int.MaxValue, err: ErrorHandler[Unit] = FailOnError[Unit]())(implicit ctx: ExecutionContext): Enumerator[T] =
65     Concurrent.unicast[T] { chan =>
66       wrappee.foldWhile({}, maxDocs)(
67         (_, res) => Cont(chan push res), errorHandler(chan, err)
68       ).
69         onComplete { case _ => chan.eofAndEnd() }
70     }
71 
72   override def bulkEnumerator(maxDocs: Int = Int.MaxValue, err: ErrorHandler[Unit] = FailOnError[Unit]())(implicit ctx: ExecutionContext): Enumerator[Iterator[T]] = Concurrent.unicast[Iterator[T]] { chan =>
73     wrappee.foldBulks({}, maxDocs)(
74       (_, bulk) => Cont(chan push bulk), errorHandler(chan, err)
75     ).
76       onComplete { case _ => chan.eofAndEnd() }
77   }
78 
79   override def responseEnumerator(maxDocs: Int = Int.MaxValue, err: ErrorHandler[Unit] = FailOnError[Unit]())(implicit ctx: ExecutionContext): Enumerator[Response] = Concurrent.unicast[Response] { chan =>
80     wrappee.foldResponses({}, maxDocs)(
81       (_, resp) => Cont(chan push resp), errorHandler(chan, err)
82     ).
83       onComplete { case _ => chan.eofAndEnd() }
84   }
85 
86 }
